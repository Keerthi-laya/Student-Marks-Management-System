students = []
def add_student():
    name = input("Enter student name: ")
    marks = int(input("Enter marks: "))
    students.append((name, marks))
    print("Student added successfully!\n")
def display_students():
    if len(students) == 0:
        print("No students available\n")
    else:
        for s in students:
            print("Name:", s[0], "Marks:", s[1])
        print()
def average_marks():
    total = 0
    for s in students:
        total += s[1]
    if len(students) > 0:
        print("Average Marks:", total / len(students))
    else:
        print("No data available")
    print()
def search_student(name, index=0):
    if index >= len(students):
        return "Student not found"
    if students[index][0] == name:
        return f"Found: {students[index]}"
    return search_student(name, index + 1)
while True:
    print("1.Add Student")
    print("2.Display Students")
    print("3.Average Marks")
    print("4.Search Student")
    print("5.Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        add_student()

    elif choice == "2":
        display_students()

    elif choice == "3":
        average_marks()

    elif choice == "4":
        name = input("Enter student name to search: ")
        print(search_student(name))
        print()

    elif choice == "5":
        print("Program Ended")
        break

    else:
        print("Invalid choice\n")
